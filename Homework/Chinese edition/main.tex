\begin{equation}\label{eq:5}
    f(y) =
    \begin{cases}
        \frac{1}{2^{k/2}\Gamma(k/2)} x^{k/2-1} e^{-x/2} & y>0              \\
        0                                               & \text{otherwise}
    \end{cases}
\end{equation}
